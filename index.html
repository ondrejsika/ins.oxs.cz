Install in one command, Debian only

Server Setup

cd && \
    apt update && \
    apt purge -y nano && \
    apt install -y vim mc htop git sudo curl wget gnupg apt-transport-https && \
    mkdir -p .ssh && \
    curl https://github.com/ondrejsika.keys >> ~/.ssh/authorized_keys

or one line

cd && apt update && apt purge -y nano && apt install -y vim mc htop git sudo curl wget gnupg apt-transport-https && mkdir -p .ssh && curl -o .ssh/authorized_keys https://github.com/ondrejsika.keys

or in script

curl -fsSL https://ins.oxs.cz/s.sh | sudo sh
